/* ============================================================
   רכיב: ריבוע יעד (מכפלה / סכום) — עזר קבוע ליד התרגיל
   ============================================================ */
function TargetSlot({label,target,state,shownValue}){
  const c = {
    idle:{bg:'#fff',border:'var(--line)',txt:'var(--ink)',badge:'var(--muted)'},
    ok:  {bg:'var(--ok-soft)',border:'var(--ok)',txt:'var(--ok)',badge:'var(--ok)'},
    err: {bg:'var(--err-soft)',border:'var(--err)',txt:'var(--err)',badge:'var(--err)'},
  }[state];
  const anim = state==='ok'?'glowOk .8s ease-out'
             : state==='err'?'shake .45s ease-in-out':'none';
  return (
    <div className="flex flex-col items-center gap-1.5" style={{minWidth:120}}>
      <span className="text-sm font-extrabold" style={{color:c.badge}}>{label}</span>
      <div className="rounded-2xl border-[3px] grid place-items-center relative"
        style={{background:c.bg,borderColor:c.border,width:116,height:84,animation:anim,
                transition:'background .35s,border-color .35s'}}>
        <span className="absolute top-1.5 left-2 text-xs font-bold math"
          style={{color:'var(--muted)'}}>= {target}</span>
        {shownValue!==null
          ? <span className="math text-3xl font-black"
              style={{color:c.txt,animation:'pop .4s ease-out'}}>{shownValue}</span>
          : <span className="text-2xl" style={{color:'var(--line)'}}>?</span>}
        {state!=='idle' && (
          <div className="absolute -top-3 -right-3 rounded-full p-1 text-white a-pop"
            style={{background:c.border}}>
            {state==='ok'?<IconCheck size={18}/>:<IconX size={18}/>}
          </div>
        )}
      </div>
    </div>
  );
}

      <div className="mt-5 bg-white rounded-2xl border overflow-hidden"
        style={{borderColor:'var(--line)'}}>
        <NavBar
          onBack={()=>setP(Math.max(0,p-1))}
          backDisabled={p===0}
          onNext={()=> isLast ? onToExam() : setP(p+1)}
          nextLabel={isLast?'למבחן המסכם':'הבא'}
          centerText={page.type==='guided'?'דוגמה מודרכת':page.type==='sign'?'משחקי סימנים':'הקדמה'}
        />
      </div>
      <Footer/>
    </div>
  );
}

/* --- עמוד דוגמה מודרכת — אינטראקטיבי, אותה מכניקה כמו המבחן --- */
function GuidedPage({data}){
  const [picked,setPicked]   = useState(null);
  const [prodState,setProd]  = useState('idle');
  const [sumState,setSum]    = useState('idle');
  const [pv,setPv]=useState(null), [sv,setSv]=useState(null);
  const [solved,setSolved]   = useState(false);
  const timers = useRef([]);

  useEffect(()=>()=>timers.current.forEach(clearTimeout),[]);
  // איפוס בעת מעבר דוגמה
  useEffect(()=>{
    timers.current.forEach(clearTimeout); timers.current=[];
    setPicked(null);setProd('idle');setSum('idle');setPv(null);setSv(null);setSolved(false);
  },[data]);

  const tryPair = (idx)=>{
    timers.current.forEach(clearTimeout); timers.current=[];
    const [a,b]=data.pairs[idx];
    setPicked(idx);
    setProd('idle');setSum('idle');setPv(null);setSv(null);setSolved(false);
    timers.current.push(setTimeout(()=>{ setPv(a*b); setProd(a*b===data.end?'ok':'err'); },550));
    timers.current.push(setTimeout(()=>{
      setSv(a+b); setSum(a+b===data.mid?'ok':'err');
      if(a*b===data.end && a+b===data.mid) setSolved(true);
    },1250));
  };

  return (
    <div className="bg-white rounded-3xl border overflow-hidden a-fade"
      style={{borderColor:'var(--line)',boxShadow:'0 12px 44px rgba(30,27,75,.07)'}}>
      <div className="p-7" style={{background:'var(--brand-soft)'}}>
        <p className="text-center text-lg font-bold mb-5">{data.story}</p>
        <TrinomialBoard exp={data.exp} mid={data.mid} end={data.end}
          prodState={prodState} sumState={sumState} prodVal={pv} sumVal={sv}/>
      </div>

      <div className="p-6">
        <p className="text-center text-base font-bold mb-3" style={{color:'var(--muted)'}}>
          נסו זוגות — לחצו וצפו אם הם ממלאים את שני הריבועים:
        </p>
        <div className="grid sm:grid-cols-3 gap-3">
          {data.pairs.map((pr,k)=>(
            <button key={k} onClick={()=>tryPair(k)}
              className="p-5 rounded-2xl border-[3px] text-2xl font-black transition-all
                hover:-translate-y-1"
              style={{borderColor:k===picked?'var(--brand)':'var(--line)',
                      background:k===picked?'var(--brand-soft)':'#fff'}}>
              <PairHe a={pr[0]} b={pr[1]}/>
            </button>
          ))}
        </div>
      </div>

      {solved && (
        <div className="px-6 pb-7 a-fade">
          <div className="p-5 rounded-2xl text-center"
            style={{background:'var(--ok-soft)',border:'2px solid var(--ok)'}}>
            <p className="text-lg font-extrabold mb-2" style={{color:'var(--ok)'}}>
              ✓ מצאתם! המספרים <PairHe a={data.answer[0]} b={data.answer[1]}/> ממלאים את שני הריבועים.
            </p>
            <div className="text-base font-bold mb-1" style={{color:'var(--muted)'}}>
              ולכן הפירוק לסוגריים הוא:
            </div>
            <div className="math text-3xl font-black" style={{color:'var(--ink)'}}>
              {data.exp} = {factorStr(data.answer[0],data.answer[1])}
            </div>
          </div>
        </div>
      )}
    </div>
  );
}

/* ============================================================
   מבחן מסכם — בלי חשיפת תשובה אוטומטית, עם הסבר דרך מלאה
   ============================================================ */
function ExamScreen({onFinish,onHome}){
  const [qi,setQi]       = useState(0);
  const [score,setScore] = useState(0);
  // מצב לכל שאלה: רשימת אינדקסים שכבר ניסה (כדי לא לחזור)
  const [tried,setTried] = useState([]);
  const [solved,setSolved] = useState(false);   // ענה נכון
  const [gaveUp,setGaveUp] = useState(false);   // ביקש לראות פתרון
  const [scored,setScored] = useState(false);   // האם כבר נספרה נקודה לשאלה הזו

  const q = examQs[qi];

  const reset = ()=>{ setTried([]); setSolved(false); setGaveUp(false); setScored(false); };

  const pick = (idx)=>{
    if(solved||gaveUp) return;
    if(tried.includes(idx)) return;
    if(idx===q.correct){
      setSolved(true);
      if(!scored){ setScore(s=>s+1); setScored(true); }
    } else {
      // תשובה שגויה: מסמנים אותה כ"נוסתה" — אבל לא חושפים את הנכונה,
      // ולא מסתירים שום אפשרות אחרת. התלמיד ממשיך לנסות.
      setTried(t=>[...t,idx]);
    }
  };

  const next = ()=>{
    if(qi<examQs.length-1){ setQi(qi+1); reset(); }
    else onFinish(score);
  };
  const back = ()=>{
    if(qi>0){ setQi(qi-1); reset(); }
  };

  const revealed = solved || gaveUp;

  return (
    <div className="max-w-3xl mx-auto px-4 py-8">
      {/* פס עליון */}
      <div className="flex items-center gap-3 mb-2">
        <button onClick={onHome} className="p-2 rounded-xl hover:bg-white transition"
          style={{color:'var(--muted)'}} title="בית"><IconHome size={20}/></button>
        <span className="text-base font-bold flex-1" style={{color:'var(--muted)'}}>
          שאלה {qi+1} מתוך {examQs.length}
        </span>
        <span className="text-base font-extrabold px-4 py-1.5 rounded-full"
          style={{background:'var(--brand-soft)',color:'var(--brand)'}}>
          ניקוד: {score}
        </span>
      </div>
      <div className="h-2.5 rounded-full mb-6 overflow-hidden" style={{background:'var(--line)'}}>
        <div className="h-full rounded-full transition-all duration-500"
          style={{width:`${((qi)/examQs.length)*100}%`,background:'var(--brand)'}}/>
      </div>

      <div className="bg-white rounded-3xl border overflow-hidden a-fade"
        style={{borderColor:'var(--line)',boxShadow:'0 12px 44px rgba(30,27,75,.08)'}}>
        {/* שאלה */}
        <div className="p-7 text-center" style={{background:'var(--brand-soft)'}}>
          <p className="text-lg font-bold mb-4">פרקו את הטרינום הבא לגורמים:</p>
          <div className="math font-black text-4xl md:text-5xl rounded-2xl bg-white border
            inline-block px-7 py-4"
            style={{borderColor:'var(--line)',color:'var(--brand-dark)'}}>{q.exp}</div>
        </div>

        {/* אפשרויות — שגויות נשארות גלויות, רק מסומנות */}
        <div className="p-6 grid sm:grid-cols-2 gap-3.5">
          {q.options.map((opt,k)=>{
            const wasTried = tried.includes(k);
            const isRightRevealed = revealed && k===q.correct;
            let st={borderColor:'var(--line)',background:'#fff',color:'var(--ink)'};
            if(wasTried)
              st={borderColor:'var(--err)',background:'var(--err-soft)',color:'var(--err)'};
            if(isRightRevealed)
              st={borderColor:'var(--ok)',background:'var(--ok-soft)',color:'var(--ok)'};
            const clickable = !revealed && !wasTried;
            return (
              <button key={k} disabled={!clickable} onClick={()=>pick(k)}
                className={`p-5 rounded-2xl border-[3px] math text-2xl font-black transition-all
                  relative ${clickable?'hover:-translate-y-1 hover:shadow-md':''}
                  ${wasTried?'a-shake':''}`}
                style={st}>
                {opt}
                {wasTried &&
                  <span className="absolute left-3 top-1/2 -translate-y-1/2"><IconX size={20}/></span>}
                {isRightRevealed &&
                  <span className="absolute left-3 top-1/2 -translate-y-1/2"><IconCheck size={20}/></span>}
              </button>
            );
          })}
        </div>

        {/* רמז עדין כשטעו — בלי לחשוף תשובה */}
        {!revealed && tried.length>0 && (
          <div className="px-6 pb-2 -mt-1">
            <p className="text-center text-base font-bold" style={{color:'var(--err)'}}>
              לא מדויק — נסו אפשרות אחרת. בדקו: מכפלה {q.end} וסכום {q.mid}.
            </p>
          </div>
        )}

        {/* כפתור "להראות פתרון" — בחירה של המורה/תלמיד, לא אוטומטי */}
        {!revealed && (
          <div className="px-6 pb-6 flex justify-center">
            <button onClick={()=>setGaveUp(true)}
              className="text-base font-bold flex items-center gap-1.5 px-4 py-2 rounded-xl
                transition hover:bg-amber-50"
              style={{color:'var(--gold)'}}>
              <IconBulb size={18}/> רוצים לראות את דרך הפתרון?
            </button>
          </div>
        )}

        {/* הסבר דרך הפתרון המלאה — מופיע רק אחרי הצלחה או בקשה מפורשת */}
        {revealed && (
          <div className="p-6 border-t a-fade" style={{borderColor:'var(--line)',
            background:solved?'var(--ok-soft)':'var(--gold-soft)'}}>
            <h4 className="font-extrabold text-lg mb-3 flex items-center gap-2"
              style={{color:solved?'var(--ok)':'var(--gold)'}}>
              {solved
                ? <><IconCheck size={22}/> כל הכבוד! הנה דרך הפתרון:</>
                : <><IconBulb size={22}/> דרך הפתרון:</>}
            </h4>
            <ol className="space-y-2.5">
              {q.steps.map((s,i)=>(
                <li key={i} className="flex gap-3 items-start a-fade"
                  style={{animationDelay:`${i*120}ms`}}>
                  <span className="shrink-0 grid place-items-center rounded-full w-7 h-7
                    text-sm font-black text-white"
                    style={{background:'var(--brand)'}}>{i+1}</span>
                  <span className="text-lg leading-relaxed pt-0.5">{s}</span>
                </li>
              ))}
            </ol>
            <div className="mt-4 text-center bg-white rounded-2xl py-3 border"
              style={{borderColor:'var(--line)'}}>
              <span className="math text-2xl font-black" style={{color:'var(--ink)'}}>
                {q.exp} = {q.options[q.correct]}
              </span>
            </div>
          </div>
        )}

        {/* ניווט קדימה/אחורה — תמיד זמין */}
        <NavBar
          onBack={back} backDisabled={qi===0}
          onNext={next}
          nextLabel={qi<examQs.length-1 ? 'לשאלה הבאה' : 'סיום ותוצאות'}
        />
      </div>
      <Footer/>
    </div>
  );
}
